player:
    id: int - id gracza
    nick: str - nick gracza
    password: str - hasło gracza
    keys: JSONB - klucze webauthn/fido2
    judge: bool - czy jest sędzia
    admin: bool - czy jest adminem
    elo: float - stan punktów
    wins: int - wygrane mecze
    loses: int - przegrane mecze

match:
    id: int - id meczu
    date: datetime - data meczu
    gamemode: enum - tryb gry (np. "8-ball", "9-ball", etc.)
    is_ranked: bool - czy mecz jest rankingowy
    additional_info: str - komentarz meczu lub dodatkowe informacje

match_players:
    id: int - id wpisu
    match_id: int - ID meczu, do którego należy wpis
    player_id: int - ID gracza, który brał udział w meczu
    is_winner: bool - True, jeśli gracz wygrał, False, jeśli przegrał
    elo_change: float - zmiana ELO gracza w tym meczu (np. +20 lub -20)
